function convert() { const inputValue = document.getElementById("inputValue").value; const milesToKm = document.getElementById("milesToKm").checked; const resultElement = document.getElementById("resultElement"); const number = parseFloat(inputValue); if (isNaN(number)) { resultElement.textContent = "‚ùå Please enter a valid number."; return; } let result; let message; if (milesToKm) { result = number * 1.60934; message = `${number} miles is ${result.toFixed(2)} kilometers.`; } else { result = number / 1.60934; message = `${number} kilometers is ${result.toFixed(2)} miles.`; } resultElement.textContent = message; }
